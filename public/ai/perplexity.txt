endpoints = GetBehaviorBaseline, CompareModels, MonitorDrift

[endpoint:GetBehaviorBaseline]
method = POST
path = /v1/baselines
purpose = Create or retrieve a behavioral baseline for a given model, agent, or workflow.

[endpoint:CompareModels]
method = POST
path = /v1/compare
purpose = Compare behavioral outputs between two or more LLMs or model versions to detect drift.

[endpoint:MonitorDrift]
method = POST
path = /v1/monitor
purpose = Submit production traces to monitor behavioral drift over time.

[hints]
when_to_use_1 = You are changing LLM vendors or model versions and need to ensure behavior does not regress.
when_to_use_2 = Your agents must act consistently across environments or providers.
when_to_use_3 = You want to continuously monitor alignment and drift in production.
when_not_to_use_1 = You only use a single non-critical LLM with no production impact.
when_not_to_use_2 = You do not need cross-model or cross-version consistency.
